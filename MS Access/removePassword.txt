
Public Sub removePassword()
    Const OLD_PASSWORD As String = "spokoloko13"
    Const NEW_PASSWORD As String = "NULL"
    Dim objConn As ADODB.Connection
    Dim strAlterPassword As String
    
    On Error GoTo ChangeDBPassword_Err
    
    ' Create the SQL string to change the database password.
    ' Here, "It" is the old password, and I am wanting to set the password to NULL
    '    Replace "It" with your password
    strAlterPassword = "ALTER DATABASE PASSWORD " & NEW_PASSWORD & " " & OLD_PASSWORD & ";"
    
    ' Open the secured database.
    Set objConn = New ADODB.Connection
    
    
 
 
    
    With objConn
        .Mode = adModeShareExclusive
        .Provider = "Microsoft.ACE.OLEDB.12.0"
        If OLD_PASSWORD <> "NULL" Then
            .Properties("Jet OLEDB:Database Password") = OLD_PASSWORD
        End If
        .Open "Data Source=\\svrark-data02\Manual Adjustment Database\EU RD\test\EU RD.mdb"
        Call .Execute(strAlterPassword)
    End With
    
    
    
    ' Clean up objects.
    objConn.Close
    Set objConn = Nothing
    

ChangeDBPassword_Err:
    MsgBox Err.Number & ":" & Err.Description
End Sub
